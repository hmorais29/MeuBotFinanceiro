# Importa as classes e funções necessárias do Flask. # request: para aceder a dados do formulário (POST). # redirect, url_for: para redirecionar o utilizador para outras rotas. # session: para gerir a sessão do utilizador (autenticação). from flask import Flask, render_template, request, redirect, url_for, session # Inicializa a aplicação Flask. app = Flask(__name__) # Configura uma chave secreta para as sessões. # ESSENCIAL para a segurança da aplicação. # Em produção, use uma chave complexa e guarde-a numa variável de ambiente. app.secret_key = 'uma_chave_secreta_muito_segura' # Rota principal - redireciona para a página de login. @app.route('/') def index(): """Redireciona o utilizador para a página de login por padrão.""" return redirect(url_for('login')) # Rota para o login do utilizador. Aceita pedidos GET e POST. @app.route('/login', methods=['GET', 'POST']) def login(): """ Exibe o formulário de login (GET) ou processa a submissão do formulário (POST). """ # Se o método for POST, o utilizador submeteu o formulário. if request.method == 'POST': # Obtém os dados do formulário. username = request.form.get('username') password = request.form.get('password') # Verificação simples das credenciais. # Numa aplicação real, usaríamos uma base de dados. if username == 'user' and password == 'pass': # Se as credenciais estiverem corretas, marca o utilizador como logado na sessão. session['logged_in'] = True session['username'] = username # Redireciona para a página do painel de controlo. return redirect(url_for('dashboard')) else: # Se as credenciais estiverem incorretas, exibe a página de login com uma mensagem de erro. return render_template('login.html', error='Credenciais inválidas.') # Se o método for GET, simplesmente exibe a página de login. return render_template('login.html') # Rota para o painel de controlo, acessível apenas se o utilizador estiver logado. @app.route('/dashboard') def dashboard(): """ Página do painel de controlo. Requer que o utilizador esteja autenticado. """ # Verifica se a chave 'logged_in' existe na sessão. if 'logged_in' in session: # Se sim, exibe a página do painel. return render_template('dashboard.html', username=session['username']) else: # Se não, redireciona o utilizador para a página de login. return redirect(url_for('login')) # Rota para o logout do utilizador. @app.route('/logout') def logout(): """Termina a sessão do utilizador e redireciona para o login.""" # Remove as chaves 'logged_in' e 'username' da sessão. session.pop('logged_in', None) session.pop('username', None) # Redireciona para a página de login. return redirect(url_for('login')) # Inicia a aplicação. if __name__ == '__main__': app.run(debug=True)
